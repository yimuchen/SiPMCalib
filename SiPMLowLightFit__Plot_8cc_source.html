202void
203SiPMLowLightFit::PrintRaw( std::ostream& sout ) const
204{
205 static const std::string rawfmt( "%10s %10.5lf %10.5lf\n" );
206
207 auto make_rooline =
208 [&rawfmt]( const std::string& name, const RooRealVar& x )->std::string {
209 return usr::fstr( rawfmt,
210 name,
211 x.getVal(),
212 x.getError() );
213 };
214 auto make_usrline =
215 [&rawfmt]( const std::string& name,
216 const usr::Measurement& x )->std::string {
217 return usr::fstr( rawfmt,
218 name,
219 x.CentralValue(),
220 x.AbsAvgError() );
221 };
222
223 sout << "Raw Fit parameter results\n";
224 sout << make_rooline( "ped", ped() );
225 sout << make_rooline( "gain", gain() );
226 sout << make_rooline( "s0", s0() );
227 sout << make_rooline( "s1", s1() );
228 sout << make_rooline( "mean", mean() );
229 sout << make_rooline( "lambda", lambda() );
230 sout << make_rooline( "alpha", alpha() );
231 sout << make_rooline( "beta", beta() );
232 sout << make_rooline( "dcfrac", dcfrac() );
233 sout << make_rooline( "epsilon", eps() );
234
235 // "Nominal" method for measuring ENF
236 const usr::Measurement enf = ExcessNoiseFactor( MeanPhotons() );
237 sout << usr::fstr( rawfmt, "ENF", enf.CentralValue(), enf.AbsAvgError() );
238
239 // "Conventional" method for measuring ENF
240 const usr::Measurement m = MeanPhotonsFromFrac();
241 const usr::Measurement enff = ExcessNoiseFactor( m );
242 sout << make_usrline( "mfrac", m );
243 sout << make_usrline( "ENFfrac", enff );
244}

247void
248SiPMLowLightFit::PrintTable( std::ostream& sout ) const
249{
250 static const std::string fmt = "%50s & = & %20s & [\\text{%s}] \\\\\n";
251 static const std::string mtitle = "\\left\\langle N_{\\gamma}\\right\\rangle";
252
253 auto mkstr = []( const usr::Measurement& x ){
254 return usr::fstr( "%.2lf \\pm %.3lf",
255 x.CentralValue(),
256 x.AbsAvgError() );
257 };
258
259 sout << "Human readable fit results\n";
260 usr::fout( fmt, mtitle, mkstr( MeanPhotons() ), "Photons" );
261 usr::fout( fmt, "\\text{Gain}", mkstr( Gain() ), "mV-ns" );
262 usr::fout( fmt, "\\sigma_\\text{com}", mkstr( CommonNoise() ), "mV-ns" );
263 usr::fout( fmt, "\\sigma_\\text{pix}", mkstr( PixelNoise() ), "mV-ns" );
264 usr::fout( fmt,
265 "P_\\text{ct}",
266 mkstr( ProbCrosstalk() * 100.0 ),
267 "\\%" );
268 usr::fout( fmt,
269 "P_\\text{ap}",
270 mkstr( ProbAfterpulse() * 100.0 ),
271 "\\%" );
272 usr::fout( fmt, "\\tau_\\text{ap}", mkstr( AfterpulseTimeNS() ), "ns" );
273 usr::fout( fmt,
274 "\\tau_\\text{dc}",
275 mkstr(
276 DarkcurrentTimeNS() / 1000.0 ),
277 "\\mus" );
278 usr::fout( fmt,
279 "ENF",
280 mkstr( ExcessNoiseFactor( MeanPhotons() ) ),
281 "" );
282}
