51WaveFormat::WaveFormat( const std::string& file, const bool invert )
52{
53 const unsigned factor = invert ?
54 -1 :
55 1;
56 std::string line;
57 std::ifstream fin( file, std::ios::in );
58
59 if( !fin.is_open() ){
60 usr::log::PrintLog( usr::log::FATAL,// Throws exception
61 usr::fstr( "Input file [%s] cannot be opened!",
62 file ) );
63 }
64
65 // Getting first line
66 std::getline( fin, line );
67 std::istringstream linestream( line );
68 linestream >> time >> nbits >> adc;
69
70 // Getting all other lines
71 while( std::getline( fin, line ) ){
72 _waveforms.push_back( std::vector<int16_t>( line.length() / nbits ) );
73
74 for( unsigned index = 0; index < _waveforms.back().size(); ++index ){
75 int16_t value = 0;
76
77 for( unsigned bit = 0; bit < nbits; ++bit ){
78 const int16_t bit_value = hex_to_int( line[nbits * index+bit] );
79 value = value << 4 | bit_value;
80 }
81
82 _waveforms.back()[index] = nbits == 4 ?
83 factor * value :
84 bit4_to_bit2( value * factor );
85 }
86
87 auto is_peak_cell = [this]( const unsigned index )->bool {
88 const auto& w = this->_waveforms.back();
89 const unsigned diffp1 = index > w.size()-2 ?
90 0 :
91 abs( w.at( index )
92 -w.at( index+1 ) );
93 const unsigned diffp2 = index > w.size()-3 ?
94 0 :
95 abs( w.at( index )
96 -w.at( index+2 ) );
97 const unsigned diffm1 = index < 1 ?
98 0 :
99 abs( w.at( index )
100 -w.at( index-1 ) );
101 const unsigned diffm2 = index < 2 ?
102 0 :
103 abs( w.at( index )
104 -w.at( index-2 ) );
105
106 const unsigned diffp = std::max( diffp1, diffp2 );
107 const unsigned diffm = std::max( diffm1, diffm2 );
108
109 return ( diffp > 70 ) && ( diffm > 70 );
110 };
111
112
113 for( unsigned index = 0; index < _waveforms.back().size(); ++index ){
114 if( is_peak_cell( index ) ){
115 _waveforms.back()[index] =
116 ( _waveforms.back()[index+1]+_waveforms.back()[index-1] ) / 2;
117 }
118 }
119 }
120}

149double
150WaveFormat::PedValue( const unsigned index,
151 const unsigned pedstart,
152 const unsigned pedstop ) const
153{
154 if( pedstart == unsigned(-1) || pedstop == unsigned(-1) ){
155 return 0;
156 }
157
158 double ped_value = 0;
159
160 for( unsigned i = pedstart; i < pedstop; ++i ){
161 ped_value += _waveforms.at( index ).at( i ) * ADC();
162 }
163
164 return ped_value / (double)( pedstop-pedstart );
165}

223double
224WaveFormat::WaveformSum( const unsigned index,
225 const unsigned intstart,
226 const unsigned intstop,
227 const unsigned pedstart,
228 const unsigned pedstop ) const
229{
230 double ans = 0;
231 const double ped_value = PedValue( index, pedstart, pedstop );
232
233 const unsigned start = std::max( intstart, (unsigned)0 );
234 const unsigned stop = std::min( intstop, NSamples() );
235
236 for( unsigned i = start; i < stop; ++i ){
237 ans += _waveforms.at( index ).at( i ) * ADC()-ped_value;
238 }
239
240 ans *= Time();
241
242 return ans;
243}
