152PairList
153MakePairs( const usr::ArgumentExtender& args,
154 const std::string& arg_name )
155{
156 std::vector<std::pair<double, double> > ans;
157 if( args.CheckArg( arg_name ) ){
158 const std::vector<double> list = args.ArgList<double>( arg_name );
159 if( list.size() % 2 == 1 ){
160 throw std::runtime_error( "Expected in put arguments to be in pairs!" );
161 }
162
163 for( unsigned i = 0; i < list.size() / 2; ++i ){
164 ans.emplace_back( list[2 * i], list[2 * i+1] );
165 }
166 }
167
168 return ans;
169}

172std::vector<ProfileMgr>
173MakeProfileList( const usr::ArgumentExtender& args,
174 const PairList& bias_pair,
175 const PairList& sipm_pair,
176 const PairList& pulse_pair )
177{
178 std::vector<ProfileMgr> ans;
179
180 if( !args.CheckArg( "biasrange" ) ){
181 for( const auto sp : sipm_pair ){
182 for( const auto pp : pulse_pair ){
183 ans.push_back( ProfileMgr() );
184 ans.back()._bias_min = 0;
185 ans.back()._bias_max = 0;
186 ans.back()._sipm_min = std::min( sp.first, sp.second );
187 ans.back()._sipm_max = std::max( sp.first, sp.second );
188 ans.back()._pulse_min = std::min( pp.first, pp.second );
189 ans.back()._pulse_max = std::max( pp.first, pp.second );
190 }
191 }
192 } else if( !args.CheckArg( "tempsipm" ) ){
193 for( const auto bp : bias_pair ){
194 for( const auto pp : pulse_pair ){
195 ans.push_back( ProfileMgr() );
196 ans.back()._bias_min = std::min( bp.first, bp.second );
197 ans.back()._bias_max = std::max( bp.first, bp.second );
198 ans.back()._sipm_min = 0;
199 ans.back()._sipm_max = 0;
200 ans.back()._pulse_min = std::min( pp.first, pp.second );
201 ans.back()._pulse_max = std::max( pp.first, pp.second );
202 }
203 }
204 } else {
205 for( const auto bp : bias_pair ){
206 for( const auto sp : sipm_pair ){
207 ans.push_back( ProfileMgr() );
208 ans.back()._bias_min = std::min( bp.first, bp.second );
209 ans.back()._bias_max = std::max( bp.first, bp.second );
210 ans.back()._sipm_min = std::min( sp.first, sp.second );
211 ans.back()._sipm_max = std::max( sp.first, sp.second );
212 }
213 }
214 }
215
216 for( auto& pm : ans ){
217 pm._profile = new TProfile( usr::RandomString( 6 ).c_str(),
218 "",
219 args.Arg<int>( "xbins" ),
220 args.Arg<double>( "xmin" ),
221 args.Arg<double>( "xmax" ) );
222 pm._profile->BuildOptions( 0, 0, "s" );
223 pm._profile->GetYaxis()->SetTitle( "Readout [V-ns]" );
224 if( pm._bias_min == pm._bias_max && pm._bias_min == 0 ){
225 pm._profile->GetXaxis()->SetTitle( "Bias Voltage [mV]" );
226 } else if( pm._sipm_min == pm._sipm_max && pm._sipm_min == 0 ){
227 pm._profile->GetXaxis()->SetTitle( "SiPM Temperature [^{#circ}C]" );
228 } else {
229 pm._profile->GetXaxis()->SetTitle( "LED Temperature [^{#circ}C]" );
230 }
231
232 if( bias_pair.size() > 1 ){
233 const double mean = ( pm._bias_min+pm._bias_max ) / 2000;
234 const double err = ( pm._bias_max-pm._bias_min ) / 2000;
235 pm._name += usr::fstr( " Bias:%.3lf#pm%.3lfV", mean, err );
236 }
237 if( sipm_pair.size() > 1 ){
238 const double mean = ( pm._sipm_min+pm._sipm_max ) / 2;
239 const double err = ( pm._sipm_max-pm._sipm_min ) / 2;
240 pm._name += usr::fstr( " SiPM:%.1lf#pm%.2lf^{#circ}C", mean, err );
241 }
242 if( pulse_pair.size() > 1 ){
243 const double mean = ( pm._pulse_min+pm._pulse_max ) / 2;
244 const double err = ( pm._pulse_max-pm._pulse_min ) / 2;
245 pm._name += usr::fstr( " LED:%.1lf#pm%.2lf^{#circ}C", mean, err );
246 }
247 }
248
249 return ans;
250}

253void
254ProfileMgr::Fill( const double readout,
255 const double bias,
256 const double ledtemp,
257 const double sipmtemp )
258{
259 if( _bias_min == _bias_max && _bias_min == 0 ){
260 if( _pulse_min <= ledtemp && ledtemp <= _pulse_max &&
261 _sipm_min <= sipmtemp && sipmtemp <= _sipm_max ){
262 _profile->Fill( bias, -readout );
263 }
264 } else if( _sipm_min == _sipm_max && _sipm_min == 0 ){
265 if( _bias_min <= bias && bias <= _bias_max && _pulse_min <= ledtemp &&
266 ledtemp <= _pulse_max ){
267 _profile->Fill( sipmtemp, -readout );
268 }
269 } else {
270 if( _bias_min <= bias && bias <= _bias_max && _sipm_min <= sipmtemp &&
271 sipmtemp <= _sipm_max ){
272 _profile->Fill( ledtemp, -readout );
273 }
274 }
