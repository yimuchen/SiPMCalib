37
38 // Calculating number of fired pixel by LO approximation
39 const double LOpar[4] = {1., pde, 0, N};
40 const double Nfired_LO = LOModel( xx, LOpar );
41
42 // Simple linear correction
43 const double NLOLinear = ( 1-alpha ) * Nfired_LO+alpha * pde * x * N;
44
45 const double nlo_frac = pde * x * N / Nfired_LO;
46 const double NonLinCorr = ( beta+1 ) / ( beta+nlo_frac );
47
48 return gain * NLOLinear * NonLinCorr+offset;
