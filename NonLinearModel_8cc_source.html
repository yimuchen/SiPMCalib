71 // Calculating number of fired pixel by LO approximation
72 const double LOpar[3] = {1.0, N, 0.0};
73 const double Nfired_LO = LOModel( xx, LOpar );
74
75 // Simple linear correction
76 const double NLOLinear = (( 1-alpha ) * Nfired_LO )+( alpha * x );
77 const double NonLinCorr = ( beta+1 ) / ( beta+( x / Nfired_LO ));
78
79 return ( gain * NLOLinear * NonLinCorr )+ped;
