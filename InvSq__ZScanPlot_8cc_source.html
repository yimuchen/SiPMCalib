152TGraphErrors
153MakeZScanGraph( const StdFormat& sformat, const double uncscale )
154{
155 std::vector<double> z = sformat.Z();
156 std::vector<double> lumi = sformat.DataCol( 0 );
157 std::vector<double> lumierr = sformat.DataCol( 1 );
158 std::vector<double> zero( z.size(), 0.0 );
159 std::vector<double> lerr_scaled( z.size(), 0.0 );
160 lerr_scaled.resize( z.size() );
161
162 std::transform( lumierr.begin(),
163 lumierr.end(),
164 lerr_scaled.begin(), [uncscale]( double x ){
165 return x * uncscale;
166 } );
167
168 return TGraphErrors( z.size(),
169 z.data(),
170 lumi.data(),
171 zero.data(),
172 lerr_scaled.data() );
173}

191TFitResult
192FitAndShiftData( TGraph& data, TF1& func, double& ped, double& zoffset )
193{
194 usr::log::PrintLog( usr::log::INFO,
195 "Getting the estimate values for fitting" );
196 const double lumimax = usr::plt::GetYmax( data );
197 const double lumimin = usr::plt::GetYmin( data );
198 const double zmin = usr::plt::GetXmin( data );
199
200 func.SetParameter( 0, 5 );
201 func.SetParameter( 1, 0 );
202 func.SetParameter( 2, ( lumimax * ( zmin+5 ) * ( zmin+5 ) ) );
203 func.SetParameter( 3, lumimin );
204 func.FixParameter( 4, 1.0 );
205
206 usr::log::PrintLog( usr::log::INFO, "Running the fit" );
207 data.Fit( &func, "Q EX0 M E N 0" ).Get();
208
209 // Saving the original fit results for the pedestal and zoffset
210 ped = func.GetParameter( 3 );
211 zoffset = func.GetParameter( 0 );
212 const double scale = func.GetParError( 0 ) / func.GetParError( 2 );
213 func.FixParameter( 4, 1 / scale );
214 func.SetParameter( 2, func.GetParameter( 2 ) * scale );
215
216 usr::log::PrintLog( usr::log::INFO, "Shifting the data for better plotting" );
217
218 for( int j = 0; j < data.GetN(); ++j ){
219 data.GetX()[j] -= zoffset;
220 data.GetY()[j] -= ped;
221 }
222
223 usr::log::PrintLog( usr::log::INFO,
224 "Shifting the function for better plotting" );
225 func.SetParameter( 0, 0.0 );
226 func.SetRange( usr::plt::GetXmin( data ), usr::plt::GetXmax( data ) );
227
228 usr::log::PrintLog( usr::log::INFO,
229 "Refitting to get a correct result container" );
230 TFitResult*ans = data.Fit( &func, "Q EX0 M E N 0 S" ).Get();
231
232 usr::log::PrintLog( usr::log::INFO, "Complete fitting routine" );
233
234 usr::log::PrintLog( usr::log::INFO,
235 usr::fstr( "Fitted z0: %.2lf +- %.2lf", zoffset,
236 func.GetParError( 0 ) ) );
237 usr::log::PrintLog( usr::log::INFO,
238 usr::fstr( "Fitted x0: %.2lf +- %.2lf",
239 func.GetParameter( 1 ),
240 func.GetParError( 1 ) ) );
241 usr::log::PrintLog( usr::log::INFO,
242 usr::fstr( "Fitted L0: %.2lf +- %.2lf",
243 func.GetParameter( 2 ),
244 func.GetParError( 2 ) ) );
245 usr::log::PrintLog( usr::log::INFO,
246 usr::fstr( "Fitted ped: %.2lf +- %.2lf", ped,
247 func.GetParError( 3 ) ) );
248
249 return *ans;
250}
