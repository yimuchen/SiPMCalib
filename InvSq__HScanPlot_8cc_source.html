18double
19ExpFunc( const double*xy, const double*param )
20{
21 const double x = xy[0];
22 const double y = xy[1];
23 const double x0 = param[0];
24 const double y0 = param[1];
25 const double z = param[2];
26 const double N = param[3];
27 const double P = param[4];
28
29 const double D2 = ( x-x0 ) * ( x-x0 )+( y-y0 ) * ( y-y0 )+z * z;
30
31 // return N/D2 + P;
32 return ( N * z ) / ( D2 * sqrt( D2 ) )+P;
33}

189TH2D*
190MakeHScanGraph( const StdFormat& sformat )
191{
192 // Getting from standard format.
193 const std::vector<double> x = sformat.X();
194 const std::vector<double> y = sformat.Y();
195 const std::vector<double> lumi = sformat.DataCol( 0 );
196 const std::vector<double> lumierr = sformat.DataCol( 1 );
197
198 // Addtional parsing.
199 const double xdiff = fabs( x.at( 0 )-x.at( 1 ) );
200 const double xmax = usr::GetMaximum( x );
201 const double xmin = usr::GetMinimum( x );
202 const double ymax = usr::GetMaximum( y );
203 const double ymin = usr::GetMinimum( y );
204 const unsigned nbins = ( xmax-xmin ) / xdiff+1;
205
206 // Creating histogram
207 TH2D*ans = new TH2D( ( "hist"+usr::RandomString(
208 6 ) ).c_str(),
209 "",
210 nbins,
211 xmin-0.5 * xdiff,
212 xmax+0.5 * xdiff,
213 nbins,
214 ymin-0.5 * xdiff,
215 ymax+0.5 * xdiff );
216
217 // Setting histogram values
218 for( unsigned i = 0; i < x.size(); ++i ){
219 const int binidx = ans->FindBin( x.at( i ), y.at( i ) );
220 ans->SetBinContent( binidx, lumi.at( i ) );
221 ans->SetBinError( binidx, lumierr.at( i ) );
222 }
223
224 ans->SetStats( 0 );
225
226 return ans;
227}
