243void
244SiPMNonLinearFit::MakeInterpolator1D()
245{
246 const auto z = _raw_data->Z();
247 const double closest_z = *min_element(
248 z.begin(),
249 z.end(), [this]( const double l, const double r )->bool {
250 return fabs( l-this->_power_z ) < fabs( r-this->_power_z );
251 } );
252 auto at_z = [closest_z]( const StdFormat::RowFormat& row )->bool {
253 return row.z == closest_z;
254 };
255 std::vector<double> bias = _raw_data->Bias( at_z );
256 std::vector<double> readout = _raw_data->DataCol( 0, at_z );
257
258 bias.insert( bias.begin(), -10000000 );
259 readout.insert( readout.begin(), readout.front() );
260 bias.push_back( 100000000 );
261 readout.push_back( readout.back() );
262
263 // Making the 1D interpolator
264 _lookup_1d.reset(
265 new ROOT::Math::Interpolator( bias,
266 readout,
267 ROOT::Math::Interpolation::kLINEAR ));
268}

282SiPMNonLinearFit::MakeInterpolator3D( const std::string& corr )
283{
284 std::ifstream infile( corr );
285 std::string line;
286
287 std::vector<double> bias;
288 std::vector<double> readout;
289
290 while( std::getline( infile, line ) ){
291 std::istringstream linestream( line );
292 double bias_value;
293 double readout_value;
294 linestream >> bias_value >> readout_value;
295 bias.push_back( bias_value );
296 readout.push_back( readout_value );
297 }
298
299 _lookup_1d.reset(
300 new ROOT::Math::Interpolator( bias,
301 readout,
302 ROOT::Math::Interpolation::kLINEAR ));
303}

312void
313SiPMNonLinearFit::MakeLinearGraph()
314{
315 auto lin_power = [this]( const StdFormat::RowFormat& row )->bool {
316 return row.bias >= this->_lin_pmin &&
317 row.bias <= this->_lin_pmax;
318 };
319 const std::vector<double> z = _raw_data->Z( lin_power );
320 const std::vector<double> lumi = _raw_data->DataCol( 0, lin_power );
321 const std::vector<double> unc = _raw_data->DataCol( 1, lin_power );
322 const std::vector<double> bias = _raw_data->Bias( lin_power );
323 const std::vector<double> stmp = _raw_data->SiPMTemp( lin_power );
324 const std::vector<double> ptmp = _raw_data->LedTemp( lin_power );
325 const std::vector<double> zero( z.size(), 0 );
326
327 assert( z.size() > 0 );
328
329 // Making the TGraph object
330 _lin_data = TGraphErrors( z.size(),
331 z.data(),
332 lumi.data(),
333 zero.data(),
334 zero.data() );
335
336 const double zmin = usr::GetMinimum( z );
337 const double zmax = usr::GetMaximum( z );
338 const double lmin = usr::GetMinimum( lumi );
339 const double lmax = usr::GetMaximum( lumi );
340
341 _lin_func = TF1( usr::RandomString( 12 ).c_str(), InvSq_Z, zmin, zmax, 5 );
342 _lin_func.FixParameter( 4, 1.0 ); // Fixing the redundent scaling paramter
343 _lin_func.SetParLimits( 3, 0, 10 );
344 _lin_data.Fit( &_lin_func, "Q EX0 M E N 0" ); // Running first fit
345
346 // Scaling scaling parameter according to uncertainty ratio
347 const double scale = _lin_func.GetParError( 0 ) / _lin_func.GetParError( 2 );
348 _lin_func.FixParameter( 4, 1 / scale );
349 _lin_func.SetParameter( 2, _lin_func.GetParameter( 2 ) * scale );
350
351 // Rerunning the fit
352 _lin_fit = *( _lin_data.Fit( &_lin_func, "Q EX0 M E N 0 S" ).Get());
353 _lin_fit = *( _lin_data.Fit( &_lin_func, "Q EX0 M E N 0 S" ).Get());
354}

385void
386SiPMNonLinearFit::MakeNonLinearGraph( const double ref_n,
387 const double ref_z,
388 const double ref_bias,
389 const double gain,
390 const double ped )
391{
392 // Finding the new reference point data power
393 const auto ref_row = *std::min_element(
394 _raw_data->begin(),
395 _raw_data->end(),
396 [ref_z, ref_bias]( const StdFormat::RowFormat& l,
397 const StdFormat::RowFormat& r )->bool {
398 if( fabs( l.z-ref_z ) != fabs( r.z-ref_z ) ){
399 return fabs( l.z-ref_z ) < fabs( r.z-ref_z );
400 } else {
401 return fabs( l.bias-ref_bias ) < fabs( r.bias-ref_bias );
402 }
403 } );
404
405 std::vector<double> n_in;
406 std::vector<double> n_out;
407 std::vector<double> in_unc;
408 std::vector<double> out_unc;
409 std::vector<double> bias;
410 std::vector<double> zeros;
411
412 // Looping over all data
413
414 // TODO: the positional estimator should also include uncertainties. Find a
415 // good way to properly propagate this.
416 for( const auto row : *_raw_data ){
417 const double z_mult = GetZPosMultiplier( row, ref_row, ped );
418 const double l_mult = GetLookUpMultiplier( row, ref_row );
419
420 n_in.push_back( ref_n * z_mult * l_mult );
421 n_out.push_back( row.data.at( 0 ) / gain );
422 in_unc.push_back( 0 ); // FIXME!!
423 out_unc.push_back( row.data.at( 1 ) / gain );
424
425 bias.push_back( row.bias );
426 zeros.push_back( 0 );
427 }
428
429 _nl_data = TGraph2DErrors( n_in.size(),
430 n_in.data(),
431 n_out.data(),
432 bias.data(),
433 in_unc.data(),
434 out_unc.data(),
435 zeros.data() );
436}

445{
446 const double xmin = usr::plt::GetXmin( _nl_data );
447 const double xmax = usr::plt::GetXmax( _nl_data );
448
449 _nl_func = TF1( usr::RandomString( 12 ).c_str(), NLOModel, xmin, xmax, 5 );
450 _nl_func.SetParameter( 0, 1.0 ); // Gain
451 _nl_func.SetParLimits( 0, 0, 1000 );
452 _nl_func.SetParameter( 2, 0.0 ); // Pedestal
453 _nl_func.SetParLimits( 2, 0.0, 100 );
454 _nl_func.SetParameter( 3, 0.0 ); // Alpha
455 _nl_func.SetParLimits( 3, 0, 1.0 );
456 _nl_func.SetParameter( 4, 0.0 ); // Beta
457 _nl_func.SetParLimits( 4, 0, 10.0 );
458
459 if( _pixel_min == _pixel_max ){
460 _nl_func.FixParameter( 1, _pixel_min );
461 } else {
462 _nl_func.SetParameter( 1, _pixel_min ); // Total Number of pixels
463 _nl_func.SetParLimits( 1, _pixel_min, _pixel_max );
464 }
465
466 TGraphErrors _nl_g = TGraphErrors( _nl_data.GetN(),
467 _nl_data.GetX(),
468 _nl_data.GetY(),
469 _nl_data.GetEX(),
470 _nl_data.GetEY() );
471
472 // Running the simple
473 _nl_fit = *( _nl_g.Fit( &_nl_func, "Q EX0 M E N 0 S" ).Get());
474
475 _nl_fit.Print();

697double
698SiPMNonLinearFit::GetZPosMultiplier( const StdFormat::RowFormat& main,
699 const StdFormat::RowFormat& ref,
700 const double ped ) const
701{
702 const double op_ped = ped == 0 ?
703 _lin_func.GetParameter( 3 ) :
704 ped;
705
706 auto eval_row = [this, op_ped]( const StdFormat::RowFormat& r )->double {
707 return ( this->_lin_func.Eval( r.z ))-op_ped;
708 };
709 return eval_row( main ) / eval_row( ref );
710}
