18void
19SiPMLowLightFit::RunPDFEstimation()
20{
21 // Resetting the objects used for estimation
22 _peakfits.clear();
23 _gain_graph.reset( nullptr );
24 _gain_fit.reset( nullptr );
25 _width_graph.reset( nullptr );
26 _width_fit.reset( nullptr );
27 _height_graph.reset( nullptr );
28 _height_fit.reset( nullptr );
29
30 _est_hist.reset( usr::TH1DFromRooData( *_data, *_x ) );
31 _spectrum.reset( new TSpectrum( 20 ) );// 20 peaks should be plenty
32 _spectrum->Search( _est_hist.get(), 1, "nobackground" );
33
34 for( int i = 0; i < _spectrum->GetNPeaks(); ++i ){
35 const double x = _spectrum->GetPositionX()[i];
36 TF1* fit = good_local_peak_fit( x );
37 if( fit ){
38 _peakfits.emplace_back( fit );
39 }
40 }
41
42 std::sort( _peakfits.begin(),
43 _peakfits.end(), []( auto& left, auto& right ){
44 return left->GetParameter( 1 ) < right->GetParameter( 1 );
45 } );
46
47 // Early exit if peak finding failed to more than 1 peak.
48 if( _peakfits.size() < 2 ){
49 usr::log::PrintLog( usr::log::WARNING,
50 "Peak finder found less than two peaks! Results from the estimations " "routine will not be used. Check to see if data is behaving properly "
51 "running as expected" );
52 return;
53 }
54
55 run_gain_est();
56 run_width_est();
57 run_height_est();
58}

61TF1*
62SiPMLowLightFit::good_local_peak_fit( const double x )
63{
64 const int bin = _est_hist->FindBin( x );
65 const double binwidth = _est_hist->GetBinWidth( bin );
66 const int maxbin = _est_hist->GetMaximumBin();
67 const double max = _est_hist->GetBinContent( maxbin );
68
69 TF1*fit = new TF1( usr::RandomString(
70 6 ).c_str(),
71 "gaus",
72 x-2 * binwidth,
73 x+2 * binwidth );
74 fit->SetParameter( 1, x );
75 fit->SetParameter( 2, 2 * binwidth );
76 _est_hist->Fit( fit, "QN0LR" );
77
78 // Checking if the peak is "good"
79 bool is_good = true;
80
81 // Ignoring peaks that are too small
82 if( _est_hist->GetBinContent( bin ) < ( max / 20. ) ){ is_good = false; }
83
84 // Skipping peaks that have maxmia outside of fit rage (not really a peak)
85 if( std::fabs( x-fit->GetParameter( 1 ) ) > 2 * binwidth ){
86 is_good = false;
87 }
88
89 // Skipping peaks that are too wide (typically intermidiate structure
90 // misidentified as primary peaks)
91 if( std::fabs( fit->GetParameter( 2 ) ) > 6 * binwidth ){ is_good = false; }
92
93 if( is_good ){
94 return fit;
95 } else {
96 delete fit;
97 return nullptr;
98 }
99}

102void
103SiPMLowLightFit::run_gain_est()
104{
105 auto f = []( const double*xx, const double*par ){
106 const double x = xx[0];
107 const double a = par[0];
108 const double b = par[1];
109 return x * a+b;
110 };
111
112 const unsigned np = _peakfits.size();
113 _gain_fit.reset( new TF1( usr::RandomString( 6 ).c_str(), f, 0, np, 2 ) );
114 _gain_graph.reset( new TGraphErrors( np ) );
115
116 for( unsigned i = 0; i < np; ++i ){
117 const double x = i;
118 const double x_err = 0;
119 const double y = _peakfits.at( i )->GetParameter( 1 );
120 const double y_err = _peakfits.at( i )->GetParError( 1 );
121
122 _gain_graph->SetPoint( i, x, y );
123 _gain_graph->SetPointError( i, x_err, y_err );
124 }
125
126 _gain_fit->SetParameter( 0, _gain_graph->GetY()[1]-_gain_graph->GetY()[0] );
127 _gain_fit->SetParameter( 1, _gain_graph->GetY()[0] );
128 _gain_graph->Fit( _gain_fit.get(), "QRN0 EX0" );
129
130 if( _gain_fit ){
131 if( !ignore_ped_est ){
132 ped() = _gain_fit->GetParameter( 1 );
133 }
134 if( !ignore_gain_est ){
135 gain() = _gain_fit->GetParameter( 0 );
136 }
137 }
138}

141void
142SiPMLowLightFit::run_width_est()
143{
144 auto f = []( const double*xx, const double*par ){
145 const double x = xx[0];
146 const double s0 = par[0];
147 const double s1 = par[1];
148 return sqrt( s0 * s0+x * s1 * s1 );
149 };
150 const unsigned np = _peakfits.size();
151 _width_fit.reset( new TF1( usr::RandomString( 6 ).c_str(), f, 0, np, 2 ) );
152 _width_graph.reset( new TGraphErrors( np ) );
153
154 for( unsigned i = 0; i < np; ++i ){
155 const double x = i;
156 const double x_err = 0;
157 const double y = _peakfits.at( i )->GetParameter( 2 );
158 const double y_err = _peakfits.at( i )->GetParError( 2 );
159
160 _width_graph->SetPoint( i, x, y );
161 _width_graph->SetPointError( i, x_err, y_err );
162 }
163
164 _width_fit->SetParameter( 0, _width_graph->GetY()[0] );
165 _width_fit->SetParameter( 1, 0 );
166 _width_graph->Fit( _width_fit.get(), "QRN0 W EX0" );
167
168 if( _width_fit ){
169 if( !ignore_s0_est ){
170 s0() = _width_fit->GetParameter( 0 );
171 }
172 if( !ignore_s1_est ){
173 s1() = _width_fit->GetParameter( 1 );
174 }
175 }
176}

179void
180SiPMLowLightFit::run_height_est()
181{
182 auto f = []( const double*xx, const double*par ){
183 const double x = xx[0];
184 const double N = par[0];
185 const double mu = par[1];
186 const double l = par[2];
187 const double y = ( mu+x * l );
188 double prod = 1;
189
190 for( int i = 1; i <= x; ++i ){
191 prod *= y;
192 prod /= (double)( i );
193 }
194
195 prod *= mu / y;
196 return N * prod * TMath::Exp( -y );
197 };
198
199 const unsigned np = _peakfits.size();
200 _height_fit.reset( new TF1( usr::RandomString( 6 ).c_str(), f, 0, np, 3 ) );
201 _height_graph.reset( new TGraphErrors( np ) );
202
203 for( unsigned i = 0; i < np; ++i ){
204 const double x = i;
205 const double x_err = 0;
206 const double y = _peakfits.at( i )->GetParameter( 0 );
207 const double y_err = _peakfits.at( i )->GetParError( 0 );
208
209 _height_graph->SetPoint( i, x, y );
210 _height_graph->SetPointError( i, x_err, y_err );
211 }
212
213 _height_fit->SetParameter( 0, _height_graph->Integral() );
214 _height_fit->SetParameter( 1, 1 );
215 _height_fit->SetParameter( 2, 0.01 );
216
217 _height_graph->Fit( _height_fit.get(), "QRN0 EX0" );
218
219 if( _height_fit ){
220 if( !ignore_mean_est ){
221 mean() = _height_fit->GetParameter( 1 );
222 }
223 if( !ignore_lambda_est ){
224 lambda() = _height_fit->GetParameter( 2 );
225 }
226 }
