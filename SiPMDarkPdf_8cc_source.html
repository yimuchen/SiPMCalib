64void
65SiPMDarkPdf::RunEstimate( const RooAbsData& data, const std::string& plot )
66{
67 TH1*h = data.createHistogram( usr::RandomString(
68 6 ).c_str(),
69 dynamic_cast<const RooAbsRealLValue&>( x.arg() ) );
70 TSpectrum spec( 5 );// 5 peaks should be plenty
71
72 spec.Search( h, 1, "nobackground" );
73
74 const double binwidth = h->GetBinWidth( 1 );
75
76 // Fitting first peak ;
77 TF1 g0( usr::RandomString( 6 ).c_str(), "gaus",
78 spec.GetPositionX()[0]-2 * binwidth,
79 spec.GetPositionX()[0]+2 * binwidth );
80 g0.SetParameter( 0, h->Integral() );
81 g0.SetParameter( 1, spec.GetPositionX()[0] );
82 g0.SetParameter( 2, 2 * binwidth );
83 h->Fit( &g0, "QN0LR" );
84
85 // Fitting the second histogram
86 TF1 g1( usr::RandomString( 6 ).c_str(), "gaus",
87 spec.GetPositionX()[1]-2 * binwidth,
88 spec.GetPositionX()[1]+2 * binwidth );
89 g1.SetParameter( 1, spec.GetPositionX()[1] );
90 g1.SetParameter( 2, 2 * binwidth );
91 h->Fit( &g1, "QN0LR" );
92
93 ped = g0.GetParameter( 1 );
94 s0 = g0.GetParameter( 2 );
95
96 s1 = sqrt( g1.GetParameter( 2 ) * g1.GetParameter( 2 )-s0 * s0 );
97 gain = g1.GetParameter( 1 )-g0.GetParameter( 1 );
98 dcfrac = g1.GetParameter( 0 ) / g0.GetParameter( 0 );
99
100 if( plot != "" ){
101 TH1D hist( usr::RandomString( 6 ).c_str(), "", h->GetNbinsX(),
102 h->GetXaxis()->GetXmin(), h->GetXaxis()->GetXmax() );
103
104 for( int i = 1; i <= h->GetNcells(); ++i ){
105 hist.SetBinContent( i, h->GetBinContent( i ) );
106 }
107
108 usr::plt::Simple1DCanvas c;
109 c.PlotHist( hist,
110 usr::plt::EntryText( "All data" ),
111 usr::plt::LineColor( usr::plt::col::black ) );
112 c.PlotFunc( g0,
113 usr::plt::EntryText(
114 "Pedestal peak fit" ),
115 usr::plt::LineColor( usr::plt::col::red ) );
116 c.PlotFunc( g1,
117 usr::plt::EntryText(
118 "1Geiger peak fit" ),
119 usr::plt::LineColor( usr::plt::col::green ) );
120
121 c.Pad().Xaxis().SetTitle( "Readout" );
122 c.Pad().Yaxis().SetTitle( "Events" );
123
124 c.SaveAsPDF( plot+"_peakfind.pdf" );
125 }
126
127 delete h;
128}
